class Solution {
public:
    vector<int> fullBloomFlowers(vector<vector<int>>& flowers, vector<int>& persons) {
        map<int,int> diff;
        for( auto & t : flowers )
        {
            diff[ t[0] ]++;
            diff[ t[1]+1 ]--;
        }


        map<int,int>::iterator it=diff.begin();
        vector<int> res( persons.size() );
        vector<int> id( persons.size() );
        //产生0，1，2，。。persons.size()
        iota( id.begin(), id.end(), 0);

        //下面是必须使用lambda表达式的场景
        //[&](int i, int j) { return persons[i] < persons[j]; }
        sort( id.begin(), id.end(),
            [&] (int a, int b)
            {
                return persons[a]<persons[b];
            }
        );

        int sum=0;
        for( auto &index : id )
        {
            while( it!=diff.end() && (*it).first <=persons[index] )
            {
                sum += (*it).second;
                ++it;
            }
            res[index]=sum;
        }

        return res;
    }
};
