            原因分析：编译器的行为！

            编译器在编译过程中，发现test的初值是可知的，并且被修饰为const，之后所有使用test的地方都以这个可预知值替换，

            • 所以才出现了，对应的汇编代码没有将test赋值给val，而是用常量5代替！

            这就是我们为什么说，只有变量才存在内存地址！常量没有地址（不包括 const定义的“伪”常量）

            # 4.对比「有this指针的成员函数调用」和「无this指针的static成员函数调用」

            #include <iostream>
#include <cstring>
using namespace std;
class demo{
public:
    int a;
    void test(){
        cout<<a<<endl;
    }
    demo(int val=7):a(val){ }
/*
				push    rbp
        mov     rbp, rsp
        mov     QWORD PTR [rbp-8], rdi
        mov     DWORD PTR [rbp-12], esi
        mov     rax, QWORD PTR [rbp-8]
        mov     edx, DWORD PTR [rbp-12]
        mov     DWORD PTR [rax], edx
        nop
        pop     rbp
        ret
*/
  
  
    static void mytest(){
        cout<<"99"<<endl;
    }
};

int main()
{
		demo whoway;
    whoway.test();
    /*
     lea     rax, [rbp-4]
     mov     rdi, rax
     call    demo::test()
    */
  
    whoway.mytest();
    //call    demo::mytest()   直接调用！没传递this指针
	return 0;
}
