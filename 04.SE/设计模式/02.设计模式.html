            # 01.单例模式（Singelton）

            # 一、背景需求

            1）概念： 单例模式（Singleton Pattern）：

            • 保证一个类仅有一个实例（只生成唯一的实例对象）
            • 并且提供一个访问它的全局访问点。

            2）项目实践中场景

            • 比如多线程编程中，要是需要通过该使用socket资源，但是明显的多次初始化socket会浪费资源，所以我们的做法是只初始化一次
            • 在Java中，servlet环境中，我们需要共享同一个资源或者操作同一个对象

            # 三、实现代码

            我们由单例模式的概念，可以抽象出我们在面向对象程序设计的时候需要这么做

            • 1）构造函数私有化
            • 2）提供一个全局访问点（用全局的静态方法实现）
            • 3）在列中定义
            • 让构造函数为 private，这样该类就不会被实例化

            根据什么时候new单例的情况，可以分为懒汉式和饿汉式

            # 单线程——懒汉式

            #include<iostream>
using namespace std;

class Singelton
{
	private:
		static Singelton * ptr;
		static int count;
	//构造函数注意是：private
		Singelton()
		{
			ptr=NULL;
			count=0;
			cout<<"调用构造函数"<<endl;
		}
	
	public:
		//获取单例的全局访问点，获取单例的接口 
		static Singelton * getInstance()
		{
			if(NULL==ptr)
			{
				ptr=new Singelton;
			} 
			return ptr;
		}
		
		static void Singelton_Print()
		{
			cout<<"cout="<<count<<endl; 
		}
};

//懒汉式的地方：并不创建单例对象 
Singelton * Singelton::ptr=NULL;
int Singelton::count=0;


int main()
{
	cout<<"单线程——懒汉式"<<endl;
	
	//调用getInstance，当单例不存在则new，存在则不new
	//保证了一个单例 
	Singelton * p1=Singelton::getInstance();
	Singelton * p2=Singelton::getInstance();
	
	if(p1!=p2) 
	{
		cout<<"不是同一个对象"<<endl; 
	}
	else
	{
		cout<<"是同一个对象"<<endl; 
	}
	
	p1->Singelton_Print();
	p2->Singelton_Print();
	
	return 0;
}

            # 单线程——饿汉式：

            #include<iostream>
using namespace std;

class Singelton
{
	private:
		static Singelton * ptr;
		static int count;
		
		Singelton()
		{
			ptr=NULL;
			count=0;
			cout<<"调用构造函数"<<endl;
		}
	
	public:
		//获取单例的全局访问点，获取单例的接口 
		static Singelton * getInstance()
		{
			
			return ptr;
		}
		
		static void Singelton_Print()
		{
			cout<<"cout="<<count<<endl; 
		}
};

//饿汉式的地方：无论你会不会用到，先给你创建了单例对象 
Singelton * Singelton::ptr=new Singelton;
int Singelton::count=0;


int main()
{
	cout<<"单线程——饿汉式"<<endl;
	
	
	Singelton * p1=Singelton::getInstance();
	Singelton * p2=Singelton::getInstance();
	
	if(p1!=p2) 
	{
		cout<<"不是同一个对象"<<endl; 
	}
	else
	{
		cout<<"是同一个对象"<<endl; 
	}
	
	p1->Singelton_Print();
	p2->Singelton_Print();
	
	return 0;
}
