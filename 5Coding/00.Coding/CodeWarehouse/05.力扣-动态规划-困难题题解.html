            • 115. 不同的子序列 (opens new window)

            dp[i][j]表示字符串s[0 ... i]的子字符串中等于字符串t[0 ... j]的子序列数目；

            初始化：

            • 1、dp[0][0] = 1，空字符串能组成空字符串；
            • 2、dp[0][1 ... m] = 0，空字符串无法组成任何非空字符串；
            • 3、dp[1 ... n][0]= 1，非空字符串可以有一种方式组成空字符串；

            当s[i] == t[j]时，分两种情况讨论：

            • 1、用s[i]去匹配t[j]，那么dp[i][j] = dp[i - 1][j - 1]；
            • 2、舍去s[i]，那么dp[i][j] = dp[i - 1][j]，总结下来就是dp[i][j] = dp[i - 1][j - 1] + dp[i - 1][j]；

            时间复杂度为O(nm)O(nm)，空间复杂度为 O(nm)O(nm)。

            作者：wo-yao-chu-qu-luan-shuo

            链接：https://leetcode.cn/problems/21dk04/solution/by-wo-yao-chu-qu-luan-shuo-n6xa/

            class Solution
{
public:
    int numDistinct(string s, string t)
    {
        if (s.length() < t.length())
        {
            return 0;
        }

        vector<vector<unsigned int>>
            dp(s.length() + 1, vector<unsigned int>(t.length() + 1, 0));
        //使用unsigned int防止溢出

        dp[0][0] = 1; //初始化1

        for (int i = 1; i <= s.length(); ++i)
        {
            //初始化2：数组元素默认已经是0，不用显式地初始化了
            dp[i][0] = 1; //初始化3
            for (int j = 1; j <= i && j <= t.length(); ++j)
            {
                if (s[i - 1] == t[j - 1])
                {
                    dp[i][j] = dp[i - 1][j - 1] + dp[i - 1][j]; //第一种情况
                }
                else
                {
                    dp[i][j] = dp[i - 1][j]; //第二种情况
                }
            }
        }

        return dp[s.length()][t.length()];
    }
};
