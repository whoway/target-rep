                                  class Solution {
public:
    int numTrees(int n) {
		vector<int> dp(n+1);
        dp[0]=1;
        dp[1]=1;
        if( n<2 ) 
        {
            return dp[n];
        }
        
        for(int cur=2; cur<=n; ++cur)
        {
            for(int LeftTreeNode=0; LeftTreeNode<cur; ++LeftTreeNode)
            {
                dp[cur] += (dp[LeftTreeNode]*dp[cur-1-LeftTreeNode]);
            }
        }
        
        return dp[n];
    }
};

                                  • F(i)为组成金额 i所需最少的硬币数量，假设在计算 F(i) 之前，我们已经计算出 F(0)到F(i-1)的答案
                                  class Solution {
public:
    int coinChange(vector<int>& coins, int amount) {
        vector<int> dp( amount+1 , 0x3f3f3f3f); //注意，初始化方式
        dp[0]=0;    //注意，初始化方式

        for(int cur=1; cur<=amount; ++cur)
        {
            //假设借助1个面值为coins[ select_index ]的硬币转移
            for(int select_index=0; select_index<coins.size(); ++select_index)
            {
                if( coins[select_index]<=cur )
                {
                    dp[cur]=min( dp[cur], dp[cur-coins[select_index] ] +1);
                }
                
            }
        }

        //如果需要的硬币数都要比全1的还多，显然不对劲
        if( dp[amount] > amount )
        {
            return -1;
        }
        else
        {
            return dp[amount];
        }
    }
};

                                  • 注意-vector的用法+这种遍历的写法！
                                  bool cmp( vector<int> a, vector<int> b)
{
    if( a[0]!=b[0] )
    {
        return a[0]<b[0];
    }

    return a[1]<b[1];
}
class Solution {
public:
    vector<vector<int>> merge(vector<vector<int>>& intervals) {
        sort( intervals.begin(), intervals.end(), cmp);
        
        vector< vector<int> > res;
        for(int i=0; i<intervals.size() ; ++i)
        {
            int cur_Left=intervals[i][0];
            int cur_Right=intervals[i][1];

            if( res.empty() )
            {
                res.push_back( intervals[i] );
            }
            else if( res.back()[1] < cur_Left )
            {
                res.push_back( intervals[i] );
            }
            else    //也就是res.back()[1] >= cur_Left，可以尝试合并，如果后边的能够比他大
            {
                res.back()[1]=max( cur_Right, res.back()[1]);
            }
        }

        return res;
    }
};

                                  • 思路来自-找左子树的最右节点，依次压平，我自己写的，但是参考思路：https://www.acwing.com/solution/LeetCode/content/202/
                                  /**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode() : val(0), left(nullptr), right(nullptr) {}
 *     TreeNode(int x) : val(x), left(nullptr), right(nullptr) {}
 *     TreeNode(int x, TreeNode *left, TreeNode *right) : val(x), left(left), right(right) {}
 * };
 */
TreeNode * searchLeftTree_right( TreeNode *root )
{
    if( nullptr==root ) 
    {
        return nullptr;
    }
    while( nullptr!=root->right )
    {
        root=root->right;
    }

    return root;
}



class Solution {
public:
    void flatten(TreeNode* root) {
        TreeNode * res=root;
        while( nullptr!=res )
        {
            TreeNode * LeftTree_right=searchLeftTree_right( res->left );
            if( nullptr==LeftTree_right )
            {
                res=res->right;
            }
            else
            {
                LeftTree_right->right=res->right;
                res->right=res->left;
                res->left=nullptr;

                res=res->right;
            } 
        }
        return ;
    }
};

                                  # 5.【中等】279. 完全平方数 (opens new window)

                                  • 动态规划
                                  class Solution {
public:
    int numSquares(int n) {
        vector<int> dp(n+1, INT_MAX);   //注意，要初始化为最大值

        //参考卡尔的：https://leetcode.cn/problems/perfect-squares/solution/dai-ma-sui-xiang-lu-279-wan-quan-ping-fa-9ieo/
        //dp[i]：和为i的完全平方数的最少数量为dp[i]
        dp[0]=0;

        for(int i=0; i<=n; ++i)
        {
            for(int del=1; i>=del*del; ++del)
            {
                dp[i]=min( dp[i], dp[i-del*del]+1 );
            }
        }

        return dp[n];
    }
};

                                  # 6.【中等】剑指 Offer II 045. 二叉树最底层最左边的值 (opens new window)

                                  • BFS的变体，简单
                                  /**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode() : val(0), left(nullptr), right(nullptr) {}
 *     TreeNode(int x) : val(x), left(nullptr), right(nullptr) {}
 *     TreeNode(int x, TreeNode *left, TreeNode *right) : val(x), left(left), right(right) {}
 * };
 */
class Solution {
public:
    int findBottomLeftValue(TreeNode* root) {
        if( nullptr==root ) return 0x3f3f3f;
        
        int res=0x3f3f3f;
        queue< TreeNode * > Q;
        Q.push( root );

        while( !Q.empty() )
        {
            res=Q.front()->val;
            int len=Q.size();
            for(int i=0; i<len; ++i)
            {
                TreeNode * cur=Q.front();
                if( nullptr!=cur->left )
                {
                    Q.push( cur->left );
                }

                if( nullptr!=cur->right )
                {
                    Q.push( cur->right );
                }

                Q.pop();
            }

        }

        return res;
    }
};

                                  # 7.【中等】128. 最长连续序列 (opens new window)

                                  • 2020-08-13有人说，字节跳动二面出现过、美团一面原题
                                    • 发现字节尤其爱问数组，一面二面全是数组
                                  • 请你设计并实现时间复杂度为 O(n) 的算法解决此问题
                                  class Solution {
public:
    int longestConsecutive(vector<int>& nums) {
        if( 0==nums.size() )
        {
            return 0;
        }
        
        unordered_map<int,bool> mp;
        for( auto val : nums)
        {
            mp[val]=true;
        }

        int res=1;
        for( auto val : nums)
        {
            int temp=1;
            if( mp[val-1] ) //防止中间数字被遍历到，算是1个减枝
            {
                continue;
            }
            else
            {
                while( mp[++val] )
                {
                    ++temp;
                }
                res=max( res, temp);
            }
        }

        return res;
    }
};

        {
            for( auto str : tail_vec )
            {
                res.push_back( string()+c+str );
            }
        }

        return res;
    }
};

                                  # 9.【中等】-水题-64. 最小路径和 (opens new window)

                                  • 水题-模仿dp数组遍历的方式
                                  • 记忆vector的2维数组的初始化方式
                                  class Solution {
public:
    int minPathSum(vector<vector<int>>& grid) {
        
        int col=grid.size();
        int row=grid[0].size();
        vector<vector<int>> dp( col, vector<int>(row) );	//基于

        dp[0][0]=grid[0][0];
        for(int i=1; i<row; ++i)
        {
            dp[0][i]=dp[0][i-1]+grid[0][i];
        }
        for(int i=1; i<col; ++i)
        {
            dp[i][0]=dp[i-1][0]+grid[i][0];
        }

        for(int i=1; i<col; ++i)
        {
            for(int j=1; j<row; ++j)
            {
                dp[i][j] = min( dp[i-1][j]+grid[i][j] , dp[i][j-1]+grid[i][j] );
            }
        }

        return dp[col-1][row-1];
    }
};

        int res=0;
        stack<int> st;
        st.push(-1);

        //「最后一个没有被匹配的右括号（注意：右边括号）的下标」

                //正好更新，新的下标为「最后一个没有被匹配的右括号（注意：右边括号）的下标」
                if(st.empty())

                    st.push(index);
                }
                else
                {
                    res=max( res, index - st.top() );
                }

            }
        }

        return res;
    }
};

public:

    void DFS_help_serialize( TreeNode * root, string &str )
    {
        if( nullptr==root )
        {
            str+="null,";	//⭐️神来之笔，借助','
            return ;
        }

        str+=( to_string(root->val) );	//int到string⭐️⭐️⭐️

        DFS_help_serialize( root->right, str);

        string str;
        DFS_help_serialize( root , str );
        return str;

    {
        //⭐️⭐️因为Q的个数必然是，完全二叉树的节点数，所以必然，不会递归到Q的size为0的情况
        if( "null"== Q.front() )
        {
            Q.pop();//一定要记得pop，不然会报错，因为多了节点⭐️⭐️⭐️
            return nullptr;
        }

        TreeNode * root=new TreeNode( stoi( Q.front() ) );  //string转换为int⭐️⭐️
        Q.pop();
        root->left=help_deserialize( Q );
        root->right=help_deserialize( Q );

        return root;

            {
                str+=c;
            }
            else
            {
                Q.push(str);
                str.clear();
            }
        }

        TreeNode * res=help_deserialize( Q );
        return res;
    }
};

bool cmp( int a[], int b[] )
{
    for(int i=0; i<26; ++i)
    {
        if( a[i]!=b[i] )
        {
            return false;
        }
    }

    return true;

public:
    vector<int> findAnagrams(string s, string p) {
        vector<int> res;
        if( p.size()>s.size() )
        {
            return res;
        }

        int index=0;
        int one[26]={0};
        int two[26]={0};
        for( auto c : p)
        {
            two[c-'a']++;
        }
        
        int Len=p.size();
        for(int i=0; i<Len; ++i)
        {
            one[ s[i]-'a' ]++;
        }
        if( cmp(one, two) )
        {
            res.push_back(index);
        }

        index=0;
        while( index+Len < s.size() )
        {
            one[ s[index]-'a']--;
            one[ s[index+Len]-'a' ]++;
            ++index;
            if( cmp(one, two) )
            {
                res.push_back(index);
            }
        }

        return res;
    }
};

                                  # 15.【中等】-406. 根据身高重建队列 (opens new window)「排序套路》⭐️

                                  • ⭐️⭐️⭐️⭐️套路？（==很难想到==）

                                  套路：一般这种数对，还涉及排序的，根据第一个元素正向排序，根据第二个元素反向排序，或者根据第一个元素反向排序，根据第二个元素正向排序，往往能够简化解题过程。

                                  参考链接：https://leetcode.cn/problems/queue-reconstruction-by-height/solution/xian-pai-xu-zai-cha-dui-dong-hua-yan-shi-suan-fa-g/

                                  • 方法

                                  我首先对数对进行排序，按照数对的元素 1 降序排序，按照数对的元素 2 升序排序。

                                  原因是，按照元素 1 进行降序排序，对于每个元素，在其之前的元素的个数，就是大于等于他的元素的数量，而按照第二个元素正向排序，我们希望 k 大的尽量在后面，减少插入操作的次数。

                                  不止是为了减少插入次数，也是为了保证正确性。 ⭐️⭐️⭐️举个例子，在身高一样，k不一样的时候，譬如[5,2]和[5,3], 对于最后排完的数组，[5,2]必然在[5,3]的前面。所以如果遍历的时候[5,3]在前面，等它先插入完，这个时候它前面会有3个大于等于它的数组对，遍历到[5,2]的时候，它必然又会插入[5,3]前面（因为它会插入链表索引为2的地方），这个时候[5,3]前面就会有4个大于等于它的数组对了，这样就会出错。

                                  bool cmp( vector<int> a, vector<int> b)
{
    if( a[0]!=b[0] )
    {
        return a[0]>b[0];   //降序
    }
    else
    {
        return a[1]<b[1];   //升序
    }
}

class Solution {
public:
    vector<vector<int>> reconstructQueue(vector<vector<int>>& people) {
        sort( people.begin(), people.end(), cmp );
        
        vector< vector<int> > res;
        for(int cur=0; cur<people.size(); ++cur)
        {
            if( people[cur][1]>=cur )
            {
                res.push_back( people[cur] );
            }
            else
            {
                int offset=people[cur][1];
                res.insert( res.begin()+offset , people[cur] );		//⭐️vector的插入
            }
        }

        return res;
    }
};

                                  # 16.【中等】-621. 任务调度器 (opens new window)-贪心+排序（脑筋急转弯）

                                  • 参考的算法来源：https://leetcode.cn/problems/task-scheduler/solution/ren-wu-diao-du-qi-by-leetcode-solution-ur9w/
                                  class Solution {
public:
    int leastInterval(vector<char>& tasks, int n) {
        if( 0==n )
        {
            return tasks.size();
        }

        
        //统计每个任务出现的次数，找到出现次数最多的任务
        vector<int> myhash(26);
        for( auto c : tasks )
        {
            myhash[ c-'A' ]++;
        }
        sort( myhash.begin(), myhash.end() );

        
        //因为相同元素必须有n个冷却时间，假设A出现3次，n = 2，任务要执行完
        //至少形成AXX AXX A序列（X看作预占位置）,,,,该序列长度为
        int minLen= (myhash[25]-1) * (n+1) + 1;

        
        //此时为了尽量利用X所预占的空间（贪心）使得整个执行序列长度尽量小，将剩余任务往X预占的空间插入
        //剩余的任务次数有两种情况：
        //1.与A出现次数相同，比如B任务最优插入结果是ABX ABX AB，中间还剩两个空位，当前序列长度+1
        //2.比A出现次数少，若还有X，则按序插入X位置，比如C出现两次，形成ABC ABC AB的序列
        //直到⭐️X预占位置还没插满⭐️，剩余元素逐个放入X位置就满足冷却时间至少为n
        for(int cur=24; cur>=0; --cur)
        {
            if( myhash[25]==myhash[cur] )
            {
                ++minLen;
            }

            if( myhash[25]>myhash[cur] )
            {
                break;
            }
        }

        //当⭐️所有X预占的位置插满⭐️了怎么办？
        //在任意插满区间（这里是ABC）后面按序插入剩余元素
        //比如ABCD ABCD发现D之间距离至少为n+1，⭐️⭐️⭐️肯定满足冷却条件
        //因此，当X预占位置能插满时，最短序列长度就是task.length，不能插满则取最少预占序列长度
        return minLen >= tasks.size() ? minLen : tasks.size(); 
    }
};

                                  # 17.【中等】-416. 分割等和子集 (opens new window)-背包「==暂时不会==」

                                  • 参考的，代码随想录：https://leetcode.cn/problems/partition-equal-subset-sum/solution/bang-ni-ba-0-1bei-bao-xue-ge-tong-tou-by-px33/
                                  class Solution {
public:
    bool canPartition(vector<int>& nums) {
        int sum = 0;

        // dp[i]中的i表示背包内总和
        // 题目中说：每个数组中的元素不会超过 100，数组的大小不会超过 200
        // 总和不会大于20000，背包最大只需要其中一半，所以10001大小就可以了
        vector<int> dp(10001, 0);
        for (int i = 0; i < nums.size(); i++) 
        {
            sum += nums[i];
        }
        if (sum % 2 == 1) 
        {
            return false;
        }
        int target = sum / 2;

        
        // 开始 01背包
        for(int i = 0; i < nums.size(); i++) {
            for(int j = target; j >= nums[i]; j--) { 
                // 每一个元素一定是不可重复放入，所以从大到小遍历
                dp[j] = max(dp[j], dp[j - nums[i]] + nums[i]);
            }
        }
        // 集合中的元素正好可以凑成总和target
        if (dp[target] == target) 
        {
            return true;
        }
        return false;
    }
};
