# 557. 反转字符串中的单词 III (opens new window)

class Solution {
public:
    string reverseWords(string s) {
        //哨兵
        s.push_back(' ');
        string res;

        int pos=0;
        while( pos<s.size() )
        {
            if( ' '==s[pos] )
            {
                res+=' ';
                ++pos;
            }
            else
            {
                int endpos=s.find(' ', pos);
                string temp=s.substr( pos, endpos-pos );
                reverse( temp.begin(), temp.end() );
                res+=temp;
                pos=endpos;
            }
        }

        res.pop_back();
        return res;
    }
};
