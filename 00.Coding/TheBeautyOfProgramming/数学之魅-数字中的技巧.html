int lowbit(uint32_t n)
{
    int sum=0;
    while(n)
    {
        ++sum;
        n&=(n-1);
    }

    return sum;
}

class Solution {
public:
    int hammingWeight(uint32_t n) {
        return lowbit(n);
    }
};

                            class Solution {
public:
    int countDigitOne(int n) {
        int res=0;
        int factorCur=1; //从个位开始『注意，为了避免，factorCur*10会溢出，要将int改为long long』
        
        while( 0!=(n/factorCur) )
        {
            
            int cur= n/factorCur%10;
            int high=n/factorCur/10;                //获得cur的左边数字
            int low = n-(n/factorCur)*factorCur;    //获得cur的右边数字

            if( 0==cur )
            {
                res+= ( high*factorCur );
            }
            else if( 1==cur )
            {
                res+=( high*factorCur+ low+1 );
            }
            else if( cur>1 ) //其实就是default
            {
                res+=( (high+1)*factorCur );
            }

            factorCur*=10;

        }

        return res;
    }
};

                            # 寻找最大的K个数

                            找到一个类似的题：

                            • 『面试题 17.14. 最小K个数 (opens new window)』

                            class Solution {
public:
    vector<int> smallestK(vector<int>& arr, int k) {
        
        vector<int> res;
        if( 0==k )
        {
            return res;
        }

        priority_queue<int> BigHeap;
        BigHeap.push( arr[0] );
        for(int use_i=1; use_i<arr.size(); ++use_i )
        {
            if( BigHeap.size()<k )
            {
                BigHeap.push( arr[use_i] );
                continue;
            }
            
            if( arr[use_i]>=BigHeap.top() )
            {
                continue;
            }
            else
            {
                BigHeap.push( arr[use_i] );
                BigHeap.pop();
            }
 
        }

        while( !BigHeap.empty() )
        {
            res.push_back( BigHeap.top() );
            BigHeap.pop();
        }

        return res;
    }
};

                            • 法2：make_heap的使用
                            class Solution {
public:
    vector<int> smallestK(vector<int>& arr, int k) {

        vector<int> rt;
        if( k<=0 )
        {
            return rt;
        }

        make_heap( arr.begin(), arr.end() );
        sort_heap( arr.begin(), arr.end() );

        for(int i=0; i<k; ++i)
        {
            rt.push_back( arr[i] );
        }

        return rt;
    }
};

                            # 求数组中最长递增子序列

                            • 『300. 最长递增子序列 (opens new window)』
                            • 未优化版的动态规划
                            

class Solution {
public:
    int lengthOfLIS(vector<int>& nums) {
        int res=0;
        if( 0==nums.size() )
        {
            return 0;
        }

        vector<int> dp( nums.size() );
        for(int loop=0; loop<nums.size(); ++loop)
        {
            dp[loop]=1; //假装，前面没有比这个更小的
            for(int before=0; before<loop; ++before)
            {
                if( nums[loop]>nums[before] )
                {
                    dp[loop]=max( dp[loop], dp[before]+1 );
                }
            }

            //cout<<loop<<":::"<<dp[loop]<<endl;
            res=max( res, dp[loop] );
        }

        return res;
    }
};
