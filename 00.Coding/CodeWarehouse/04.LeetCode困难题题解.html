                                class Solution {
public:
    string makeLargestSpecial(string s) {
        if( s.size()<=2 )
        {
            return s;
        }

        vector<string> subs;

        int count=0,Left=0;
        for(int index=0; index<s.size(); ++index)
        {
            '1'==s[index] ? ++count : --count;

            if( 0==count )
            {
                //递归分解
                string str = "1"+makeLargestSpecial( s.substr(Left+1, index-(Left+1) ) )+"0";
                subs.push_back( str );

                //更新下1个可能的特殊序列开头
                Left=index+1;
            }
        }

        sort( subs.begin(), subs.end(), greater<string>() );

        return accumulate( subs.begin(), subs.end(), string() );
    }
};

                                # 1.2.关于accumulate(subs.begin(), subs.end(), ""s);的""s解释

                                • 参考文章：关于c ++：为什么隐式转换无法累计？ (opens new window)

                                我想在通用STL函数accumulate的调用中使用从const char *到string的隐式转换。 我知道从const char *到字符串的转换不是显式的，因此我们可以将const char *参数传递给需要string类型的调用。 这可以通过上面的test_string函数来证明。 但是，当我在accumulate中执行相同的操作时，编译器会抱怨：

                                error C2440: '=': cannot convert from 'std::basic_string<char,std::char_traits<char>,std::allocator<char>>' to 'const char *'
                                1
                                • 该代码仅在将""替换为string("")时有效。 我不明白为什么隐式转换对我的自定义函数有效，但在accumulate中却不起作用。 你能解释一下吗？ 非常感谢。
                                • "我想使用从const char *到string的隐式转换"。但是编译器不知道您"想要"那样。 表达式accumulate(v.cbegin(), v.cend(),"")中绝对没有任何内容暗示应将""隐式转换为std::string。 您为什么期望编译器执行转换？ 您如何期望编译器确定转换是必需的？
                                • 您可以显式传递std::string，也可以显式指定模板参数。例如
                                // pass a std::string exactly
string sum = accumulate(v.cbegin(), v.cend(), string(""));

// T is specified as std::string explicitly
//"" will be implicitly converted to std::string
string sum = accumulate<decltype(v.cbegin()), string>(v.cbegin(), v.cend(),"");

                                class Solution {
public:
    string makeLargestSpecial(string s) {
        if (s.size() <= 2) {
            return s;
        }
        int cnt = 0, left = 0;
        vector<string> subs;
        for (int i = 0; i < s.size(); ++i) {
            if (s[i] == '1') {
                ++cnt;
            }
            else {
                --cnt;
                if (cnt == 0) {
                    subs.push_back("1" + makeLargestSpecial(s.substr(left + 1, i - left - 1)) + "0");
                    left = i + 1;
                }
            }
        }

        sort(subs.begin(), subs.end(), greater<string>{});
        string ans = accumulate(subs.begin(), subs.end(), ""s);
        return ans;
    }
};

                                class Solution {
public:
    int candy(vector<int>& ratings) {
        int n = ratings.size();
        vector<int> left(n);

        //左规则
        for (int i = 0; i < n; i++) 
        {
            if (i > 0 && ratings[i - 1]< ratings[i] ) 
            {
                left[i] = left[i - 1] + 1;
            } else 
            {
                left[i] = 1;
            }
        }

        //右规则
        int right = 0, ret = 0;
        for (int i = n - 1; i >= 0; i--) 
        {
            if (i < n - 1 && ratings[i] > ratings[i + 1]) 
            {
                right++;
            } else 
            {
                right = 1;
            }

            ret += max(left[i], right);
        }
        return ret;
    }
};

                                class Solution {
public:
    int calculate(string s) {
        stack<int> ops;
        ops.push(1);
        int sign = 1;

        int ret = 0;
        int n = s.length();
        int i = 0;
        while (i < n) {
            if (s[i] == ' ') {
                i++;
            } else if (s[i] == '+') {
                sign = ops.top();
                i++;
            } else if (s[i] == '-') {
                sign = -ops.top();
                i++;
            } else if (s[i] == '(') {
                ops.push(sign);
                i++;
            } else if (s[i] == ')') {
                ops.pop();
                i++;
            } else {
                long num = 0;
                while (i < n && isdigit( s[i] ) ) {
                    num = num * 10 + s[i] - '0';
                    i++;
                }
                ret += sign * num;
            }
        }
        return ret;
    }
};

                                # 1.解数独

                                • 好像以前，华为笔试考过
                                • https://leetcode.cn/problems/sudoku-solver/

                                参考：回溯和遍历，宫水的题解：https://mp.weixin.qq.com/s/0y4lGAH43U3w5loTgaeyoQ

                                bool hash_row[9][9];//9行，标记某行是否出现了数字1-9
bool hash_col[9][9];
bool hash_cell[3][3][9];//单元格，标记某个位置的单元格。是否出现了数组1-9


//对每一个需要填入数字的位置进行填入，如果发现填入某个数会导致数独解不下去，则进行回溯：
bool DFS(vector<vector<char>>& board, int x, int y)
{
    if( 9==y )  //列的边界
    {
        return DFS( board, x+1, 0);
    }
    if( 9==x  )//递归边界
    {
        return true;
    }

    //依次，每行的去递归
    if( '.'!=board[x][y] )
    {
        return DFS(board, x, y+1);
    }

    //依次尝试‘1’-‘9’,但是我们使用偏移量
    for(int offset=0; offset<9; ++offset)
    {
        //如果该数字在行，列，cell中都没出现过
        if( !hash_row[x][offset] && !hash_col[y][offset] && !hash_cell[x/3][y/3][offset] )
        {
            board[x][y]= offset+'1';
            hash_row[x][ offset ] =true;
            hash_col[y][ offset ] =true;
            hash_cell[x/3][y/3][ offset ]= true;

            //如果这样填后，后边的也可以继续，那就可行
            if( DFS(board, x, y+1) )
            {
                break;
            }
            else    //否则复原，后续去回溯
            {
                board[x][y]='.';
                hash_row[x][ offset ] =false;
                hash_col[y][ offset ] =false;
                hash_cell[x/3][y/3][ offset ]= false;
            }
        }
    }

    return '.'!=board[x][y];    //如果尝试后，可以填，那就true，否则false表示无解
}

class Solution {
public:
    void solveSudoku(vector<vector<char>>& board) {
        memset( hash_row, 0, sizeof(hash_row));
        memset( hash_col, 0, sizeof(hash_col));
        memset( hash_cell, 0, sizeof(hash_cell));

        //哈希标记，然后再DFS去回溯
        for(int i=0; i<9; ++i)
        {
            for(int j=0; j<9; ++j)
            {
                if( '.'!=board[i][j] )
                {
                    int index=board[i][j]-'1';//注意，偏移量是相对于'1'的
                    hash_row[i][ index ] =true;
                    hash_col[j][ index ] =true;
                    hash_cell[i/3][j/3][ index ]= true; //向下取整，获得单元格位置
                }
            }
        }

        DFS( board, 0, 0);
    }
};

                                # K个一组反转链表

                                • B站
                                /**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode() : val(0), next(nullptr) {}
 *     ListNode(int x) : val(x), next(nullptr) {}
 *     ListNode(int x, ListNode *next) : val(x), next(next) {}
 * };
 */

ListNode * reverse( ListNode * head )
{
    ListNode * sentry=nullptr;

    while( nullptr!=head )
    {
        ListNode * temp=head->next;
        head->next=sentry;
        sentry=head;
        head=temp;
    }

    return sentry;
}
class Solution {
public:
    ListNode* reverseKGroup(ListNode* head, int k) {
        if( nullptr==head || nullptr==head->next ) return head;

        ListNode * dummy=new ListNode(0);//哨兵
        dummy->next=head;

        //核心4个节点
        ListNode * pre=dummy;
        ListNode * begin=head;
        ListNode * end=head;
        ListNode * next=head->next;
        while( nullptr!=next )
        {
            for(int i=1 ; i<k && nullptr!=end; ++i)
            {
                end=end->next;
            }

            //不足K个
            if( nullptr==end )
            {
                break;
            }

            next=end->next; //更新后续「大块的节点」
            end->next=nullptr;//为调用局部的翻转做准备
            end=begin;  //方便后续连接
            begin=reverse( begin ); //翻转

            end->next=next;
            pre->next=begin;

            pre=end;
            begin=next;
            end=begin;


        }

        return dummy->next;
        
    }
};

                                # 剑指 Offer 19. 正则表达式匹配 (opens new window)

                                • 10. 正则表达式匹配 (opens new window)
                                class Solution {
public:
    bool isMatch(string s, string p) {
        int m = s.size();
        int n = p.size();

        
        auto matches = [&](int i, int j) {
            if (i == 0) {
                return false;
            }
            if (p[j - 1] == '.') {
                return true;
            }
            return s[i - 1] == p[j - 1];
        };

        vector<vector<int>> f(m + 1, vector<int>(n + 1));
        f[0][0] = true;
        for (int i = 0; i <= m; ++i) {
            for (int j = 1; j <= n; ++j) {
                if (p[j - 1] == '*') {
                    f[i][j] |= f[i][j - 2];
                    if (matches(i, j - 1)) {
                        f[i][j] |= f[i - 1][j];
                    }
                }
                else {
                    if (matches(i, j)) {
                        f[i][j] |= f[i - 1][j - 1];
                    }
                }
            }
        }
        return f[m][n];
    }
};

                                class Solution {
public:
    int minimumOneBitOperations(int n) {
        int res=0;

        while( n )
        {
            res^=n;

            n>>=1;
        }

        return res;
    }
};

                                class Solution {
public:
    int maxSatisfaction(vector<int>& satisfaction) {
        int res=0;
        int prefix_sum=0;
        sort(satisfaction.begin(), satisfaction.end(), greater<int>() );
        
        for( auto & val : satisfaction )
        {
            //贪心策略
            if( prefix_sum+val>0 )
            {
                prefix_sum+=val; //更新为s0+s1+s2...sn的形式
                res+=prefix_sum; //贪心，那么(s0)+(s0+s1)+(s0+s1+s2)类似的
            }
            else
            {
                break;
            }
        }

        return res;
    }
};

            }
        }

        return res;
    }

                                # 4.1.思路

                                • 找到两个坐标，i和j，使得flowers[i] == flowers[j]并且他们之间的和最大
                                  • 这个通过，我们的哈希mp去找最右边的位置
                                • 这两个数之间的负数也可以去除
                                  • 通过前缀和的方式去除
                                  • //前缀和累加所有正数的和，比如prefix[cur+1]，表示从prefix[0]到prefix[cur]的正数和
                                • 两个相等的数加上他们之间的正数和最大
                                  • 为了让区间有效值最大，计算前缀和时，只有元素值为正数时才进行累加， 若元素值为负，则不累加。
                                class Solution {
public:
    int maximumBeauty(vector<int>& flowers) {
        unordered_map<int ,int> mp;

        int res = INT_MIN;
        int n = flowers.size();

        vector<int> prefix(n + 1, 0); 
        //前缀和累加所有正数的和，比如prefix[cur+1]，表示从prefix[0]到prefix[cur]的正数和

        for (int i = 0; i < n; i++) 
        {
            mp[ flowers[i] ] = i; //获取本元素，最后1个出现的下标 

            prefix[i + 1] = prefix[i] + max( flowers[i] , 0 );
        }

        for (int i = 0; i < n; i++) 
        {
            //如果不存在元素flowers[i]
            if ( ! mp.count( flowers[i] ) ) 
            {
                continue;
            }

            if (i != mp[flowers[i]] ) 
            {
                //通过前缀和，更新res
                res = max( res, 2*flowers[i] + 
                          prefix[ mp[flowers[i]] ] - prefix[i + 1] ) ;
            }
        }

        return res;
    }
};

                                # 6.765. 情侣牵手 (opens new window)

                                • 方法1：贪心，复杂度为O(n^2)，但是本题还能「并查集」
                                • 该题可使用贪心思想进行操作，从题目中得到情侣们按顺序编号，都是一个偶数搭配一个奇数组成一对情侣。
                                  • 所以若当前遍历的值为偶数或 0，则它所对应的情侣应该是当前元素值 + 1
                                  • 若当前遍历的值为奇数，则它所对应的情侣应该是当前元素值 - 1。
                                  • 贪心的操作步骤如下：
                                  • (1) 首先从 row 数组的第一个元素开始判断
                                    • 若是偶数或 0，则向后遍历找到 row[i] + 1 的值保存下来与 row[i + 1] 交换位置----changeNum + 1
                                    • 若是奇数，则向后遍历找到 row[i] - 1 的值保存下来与 row[i + 1] 交换位置----changeNum + 1
                                    • (2) 交换完后当前 i 位置上的情侣已经配对。
                                int find_index( int start , const vector<int> & arr , int val)
{
    int res=start;
    for(int i=start; i<arr.size(); ++i)
    {
        if( val==arr[i] )
        {
            res=i;
            break;
        }
    }
    return res;
}

class Solution {
public:
    int minSwapsCouples(vector<int>& row) {
        int res=0;

        for(int index=0; index<row.size()-1; index+=2 )
        {
            if( row[index] & 1 )
            {
                int temp=find_index( index+1 , row, row[index]-1 );
                if( temp!=index+1 )
                {
                    ++res;
                    swap( row[index+1], row[temp] );
                }
            }
            else
            {
                int temp=find_index( index+1 , row, row[index]+1 );
                if( temp!=index+1 )
                {
                    ++res;
                    swap( row[index+1], row[temp] );
                }
            }

        }

        return res;
    }
};

                                # 7.2136. 全部开花的最早一天 (opens new window)

                                贪心算法。

                                • 第一，让某种植物种植一气呵成，不要交错种植不同植物。
                                • 第二，优先种植生长时间长的植物（就是开花慢的），确保在更多的生长期进行种植。
                                bool cmp( pair<int,int> & a , pair<int,int> &b )
{
    return a.second>b.second;
}
class Solution {
public:
    int earliestFullBloom(vector<int>& plantTime, vector<int>& growTime) {
        vector< pair<int,int> > solve;
        for( int index=0; index<plantTime.size(); ++index )
        {
            solve.push_back( make_pair( plantTime[index], growTime[index] ) );
        }

        sort( solve.begin(), solve.end() , cmp );
        int res=0;
        int prefix=0;  //到目前为止，我们一定要花费的，播种的时间
        for( auto & temp : solve )
        {
            prefix+=temp.first;
            res=max( res, prefix + temp.second ); //我们要关注生长时间会不会导致最终的总时间变长
        }

        return res;
    }
};
