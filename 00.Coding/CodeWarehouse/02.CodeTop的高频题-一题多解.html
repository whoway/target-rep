                class Solution {
public:
    string reverseWords(string s) {
        string res;
        stack<string> vec;
        queue<char> help;
        for( auto c : s)
        {
            if( ' '==c )
            {
                string temp;
                while( !help.empty() )
                {
                    temp+=help.front();
                    help.pop();
                }
                if( 0!=temp.size() )
                {
                    vec.push( temp );
                }

            }
            else
            {
                help.push( c );
            }
        }

        string temp;
        while( !help.empty() )
        {
            temp+=help.front();
            help.pop();
        }
        if( 0!=temp.size() )
        {
            vec.push( temp );
        }


        while( !vec.empty() )
        {
            res+=vec.top();
            res+=' ';

            vec.pop();
        }
       
        res.pop_back();
        return res;
    }
};

                # 方法2-原地解法⭐️⭐️

                • **进阶：**如果字符串在你使用的编程语言中是一种可变数据类型，请尝试使用 O(1) 额外空间复杂度的 原地 解法。
                • 参考自，LeetCode题解 (opens new window)
                class Solution {
public:
    string reverseWords(string s) {
        // 反转整个字符串
        reverse(s.begin(), s.end());

        int n = s.size();
        int idx = 0;
        for (int start = 0; start < n; ++start) {
            if (s[start] != ' ') {
                // 填一个空白字符然后将idx移动到下一个单词的开头位置
                if (idx != 0) s[idx++] = ' ';

                // 循环遍历至单词的末尾
                int end = start;
                while (end < n && s[end] != ' ') s[idx++] = s[end++];

                // 反转整个单词
                reverse(s.begin() + idx - (end - start), s.begin() + idx);

                // 更新start，去找下一个单词
                start = end;
            }
        }
        s.erase(s.begin() + idx, s.end());
        return s;
    }
};

                # 2.162. 寻找峰值 (opens new window)

                • 时间复杂度O(n)
                class Solution {
public:
    int findPeakElement(vector<int>& nums) {
        if( 1==nums.size() )
        {
            return 0;
        }

        if( nums[0]>nums[1] )
        {
            return 0;
        }
        if( nums[ nums.size()-1] > nums[ nums.size()-2] )
        {
            return nums.size()-1;
        }

        for(int index=1; index < nums.size()-1; ++index)
        {
            if( nums[index] > nums[index-1]  && nums[index] > nums[index+1] )
            {
                return index;
            }
        }

        return 0;
    }
};

                # 3.958. 二叉树的完全性检验 (opens new window)

                • BFS变体题，借助null哑指针
                /**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode() : val(0), left(nullptr), right(nullptr) {}
 *     TreeNode(int x) : val(x), left(nullptr), right(nullptr) {}
 *     TreeNode(int x, TreeNode *left, TreeNode *right) : val(x), left(left), right(right) {}
 * };
 */

vector<int> help;
//借助BFS改良版，来使用null这种空指针来辅助判断
void BFS( TreeNode * root)
{
    if( nullptr==root )
    {
        help.push_back( -1 );
        return ;
    }
    queue< TreeNode * > Q;
    Q.push( root );

    while( !Q.empty() )
    {
        int Len=Q.size();
        for(int i=0; i<Len; ++i)
        {
            TreeNode * cur=Q.front();
            if( nullptr==cur )
            {
                help.push_back( -1 );
                Q.pop();
                continue;
            }

            //nullptr!=cur
            help.push_back( cur->val );
            Q.pop();
            Q.push( cur->left );
            Q.push( cur->right );
        }
    }

}

class Solution {
public:
    bool isCompleteTree(TreeNode* root) {
        if( nullptr==root )
        {
            return true;
        }

        help.clear();
        BFS( root );

        int Left=0;
        int Right=help.size()-1;
        while( -1==help[Right] )
        {
            --Right;
        }

        for(int index=Left; index<Right; ++index)
        {
            if( -1==help[index] )
            {
                return false;
            }
        }

        return true;
    }
};

                # 4.74. 搜索二维矩阵 (opens new window)「不要重做」

                • 遍历的-思路要奇怪，但是感觉做了好多次了
                class Solution {
public:
    bool searchMatrix(vector<vector<int>>& matrix, int target) {
        int col=matrix.size();
        int row=matrix[0].size();

        int cur_col=col-1;
        int cur_row=0;

        while( 0<=cur_col && cur_col<col && cur_row>=0 && cur_row<row)
        {
            if( target==matrix[ cur_col ][ cur_row] )
            {
                return true;
            }
            else if( target < matrix[ cur_col ][ cur_row] )
            {
                --cur_col;
            }
            else
            {
                ++cur_row;
            }
        }

        return false;
    }
};

                # 5.670. 最大交换 (opens new window)

                后缀数组+贪心

                • 1、首先使用，从后往前遍历，后缀数组表示，从后往前，获得比当前数字更大或相等的数字的位置的第1次出现位置（以保证出现1993这样的样例时候，不会更换成9193，而应该是9913）
                • 2、获得后缀数组后，我们从前往后遍历，对首个后缀数组表示，当前数字后边有更大或相等的数字的数字进行尝试swap
                  • 1.1.如果后缀数组导向的数字和当前一样大，那我们继续从前往后遍历，比如9193，不能交换9和9，应该用1和9交换，最后9913
                  • 1.2.如果后缀数组导向的数字比当前的大（又由于1中保证了，会是偏后边的数字和我们交换）我们至此交换后break，获得的就是最后的结果。
                class Solution {
public:
    int maximumSwap(int num) {
        char temp[10];
        sprintf( temp , "%d", num);
        int len=0;

        while( '\0'!=temp[len] )
        {
            ++len;
        }

        //0x3f3f3f3f表示，右边没有比本数字更大的数字，所以下标为0x3f3f3f3f
        vector<int> up_than_cur_num_index(len, 0x3f3f3f3f);
        for(int loop=len-2; loop>=0; --loop)
        {
            int index=loop+1;
            if( 0x3f3f3f3f!=up_than_cur_num_index[loop+1] )
            {
                index=up_than_cur_num_index[loop+1];
            }

            //⭐️注意，使用>=而不是>，这样获取比如1993，这样的，右边第1个大于等于的数字，而不是纯粹大的数字
            if( temp[index]-'0' >= temp[loop]-'0' )
            {
                up_than_cur_num_index[loop]=index;
            }
        }

        for(int cur=0; cur<len-1; ++cur)
        {
            //⭐️易错，比如8183,必须要判断，可能的，8和8交换
            if( 0x3f3f3f3f!=up_than_cur_num_index[cur] 
               && temp[cur]!=temp[ up_than_cur_num_index[cur] ] )
            {
                swap( temp[cur] , temp[ up_than_cur_num_index[cur] ] );
                break;
            }
        }

        int res=0;
        for(int i=0; i<len; ++i)
        {
            res=res*10+(temp[i]-'0');
        }

        return res;
    }
};

                # 6.622. 设计循环队列 (opens new window)

                循环队列，讲解很好，参考资料 (opens new window)

                • 循环队列中，当队列为空，可知 front_loc == rear_loc

                • 而当所有队列空间全占满时，也有 front_loc == rear_loc。

                  • 为了区别⭐️⭐️这两种情况，假设队列使用的数组有 Len个存储空间，则此时规定循环队列最多只能有Len−1 个队列元素

                • 当循环队列中只剩下一个空存储单元时，则表示队列已满。

                  • 根据以上可知，队列判空的条件是 front_loc == rear_loc

                  • 而队列判满的条件是front_loc == (rear_loc+1) % Len

                • 一定要⭐️注意，维护front_loc 和rear_loc总是在Len区间⭐️

                class MyCircularQueue {
private:
    vector<int> vec;
    int Len;
    int front_loc;  //表示数组第1个元素存放位置，并且记得维护front_loc 在Len区间
    int rear_loc;
public:
    MyCircularQueue(int k) {
        vec.resize( k+1 );  //开k+1但是只存k个元素
        Len=k+1;
        front_loc=0;
        rear_loc=0;
    }
    
    bool enQueue(int value) {
        if( !isFull() )   //判断队列是否满了
        {
            vec[ rear_loc ] = value;
            rear_loc++;
            rear_loc%=Len;
            return true;
        }

        return false;
    }
    
    bool deQueue() {
        if( isEmpty() )
        {
            return false;
        }

        front_loc=(front_loc+1)%Len;
        return true;
    }
    
    int Front() {
        if( isEmpty() )
        {
            return -1;
        }

        return vec[ front_loc ];
    }
    
    int Rear() {
        if( isEmpty() )
        {
            return -1;
        }

        return vec[ (rear_loc-1+Len)%Len  ];
    }
    
    bool isEmpty() {

        if( front_loc == rear_loc )
        {
            return true;
        }
        return false;
    }
    
    bool isFull() {
        if( front_loc == (rear_loc+1)%Len )
        {
            return true;
        }

        return false;
    }
};

 * int param_3 = obj->Front();
 * int param_4 = obj->Rear();

                # 困难-899. 有序队列 (opens new window)

                • 参考的题解 (opens new window)
                • 下面的题解，来自，巧妙的轮转思维（转盘 (opens new window)），下面的图很牛

                当 k = 2 时： 我们将字符串第一个位置 s[0] 拎出来; 剩下的所有位置看作一个轮盘，通过不断将字符串第二个位置 s[1] 放到字符串末尾，即可实现将轮盘转动起来； 那么可以在适当的时候将 s[0] 插入轮盘，也可以在适当的时候将 s[1] 拎出来放到位置 s[0] 处，通过这么考虑我们即可实现对 s 字符串的排序。

                    
class Solution {
public:
    string orderlyQueue(string s, int k) {
        if( 1==k )
        {
            string res=s;
            int loop=s.size();
            while(loop--)
            {
                char c=s[0];
                s=s.substr(1, s.size()-1);
                s+=c;

                res=min( res, s);
            }

            return res;
        }
        else
        {
            //k>1的情况，总是能获得最小的字典
            sort( s.begin(), s.end() );
            return s;
        }
    }
};
