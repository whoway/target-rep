                                        unordered_map<int,char> mp;
void init()
{
    for(int index=1; index<=26; ++index)
    {
        mp[index]= (index-1+'A');
    }
    
}
class Solution {
public:
    string convertToTitle(int columnNumber) {
        init();
        stack<int> st;

        while( columnNumber )
        {
            
            columnNumber--; 
            //本题，最核心的坑：一般性的进制转换题目无须进行额外操作
            //是因为我们是在「每一位数值范围在 [0,x)[0,x)」的前提下进行「逢 xx 进一」。
        //但本题需要我们将从 11 开始，因此在执行「进制转换」操作前，我们需要先对
        // columnNumbercolumnNumber 执行减一操作，从而实现整体偏移。
            st.push( columnNumber%26 + 1);
            columnNumber/=26;
        }

        string res;
        while( !st.empty() )
        {
            res+=mp[ st.top() ];
            st.pop();
        }

        return res;
    }
};

                                        # 2.1.257. 二叉树的所有路径 (opens new window)

                                        • 递归时候临时和引用的区别，一般的题，有注意事项
                                        /**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode() : val(0), left(nullptr), right(nullptr) {}
 *     TreeNode(int x) : val(x), left(nullptr), right(nullptr) {}
 *     TreeNode(int x, TreeNode *left, TreeNode *right) : val(x), left(left), right(right) {}
 * };
 */
vector<string> res;
void DFS(TreeNode * root, string str)
{
    if(nullptr==root)
    {
        return ;
    }
    if( nullptr==root->left && nullptr==root->right)
    {
        str+="->";
        str+=to_string(root->val);

        str=str.substr(2, str.size()-2); //⭐️去除前置的->⭐️
        res.push_back( str );
        return ;
    }

    if( nullptr!=root->left )
    {
        //注意。⭐️使用临时变量，防止影响后边递归右边⭐️
        string temp=str+"->"+to_string(root->val);
        DFS( root->left , temp );
    }

    if( nullptr!=root->right )
    {
        string temp=str+"->"+to_string(root->val);
        DFS( root->right , temp );
    }
}

class Solution {
public:
    vector<string> binaryTreePaths(TreeNode* root) {
        res.clear();
        if( nullptr==root->left && nullptr==root->right )
        {
            res.push_back( to_string(root->val) );
            return res;
        }
        
        DFS( root, string( ) );

        return res;
    }
};

                                        # 2.2.1022. 从根到叶的二进制数之和 (opens new window)

                                        • DFS+递归使用临时变量的话，前置不影响，水题-递归
                                        /**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode() : val(0), left(nullptr), right(nullptr) {}
 *     TreeNode(int x) : val(x), left(nullptr), right(nullptr) {}
 *     TreeNode(int x, TreeNode *left, TreeNode *right) : val(x), left(left), right(right) {}
 * };
 */
int res;
void DFS( TreeNode * root, int cur_num)
{
    if( nullptr==root )
    {
        return ;
    }
    
    if( nullptr==root->left && nullptr==root->right )
    {
        cur_num=cur_num*2+root->val;
        res+=cur_num;
        return ;
    }

    //使用⭐️临时变量，cur_num*2+root->val，这样就⭐️⭐️不会影响后续结果
    DFS( root->left , cur_num*2+root->val);

    DFS( root->right, cur_num*2+root->val);
}
class Solution {
public:
    int sumRootToLeaf(TreeNode* root) {
        res=0;
        DFS( root , 0);
        return res;
    }
};

vector<string> res;
void DFS(string &s, int cur, string temp)
{
    if( cur==s.size() )
    {
        res.push_back( temp );
        return ;
    }

    if( !isdigit(s[cur]) )
    {
        string lower= temp; //递归，使用临时变量，防止影响
        lower+=tolower(s[cur]);
        DFS(s,cur+1,lower );
        

        string upper= temp;
        upper+=toupper(s[cur]);
        DFS(s,cur+1, upper );
    }
    else
    {
        DFS( s, cur+1, temp+=s[cur] );
    }
}
class Solution {
public:
    vector<string> letterCasePermutation(string s) {
        res.clear();
        DFS( s ,0, string() );
        return res;
    }
};

        int index=num.size()-1;

        {

                                        # 4.闰年模板-1154. 一年中的第几天 (opens new window)

                                        • 参考题目：1154. 一年中的第几天 (opens new window)

                                        闰年规则，摘自百度百科：

                                        普通年份能被4整除，且不能被100整除的，是闰年。（ 如2004年就是闰年） 世纪年份能被400整除的是闰年。（ 如2000年是闰年，1900年不是闰年） 对于数值很大的年份，这年如果能被3200整除，并且还能被172800整除的才是闰年。

                                        作者：wfnuser 链接：https://leetcode.cn/problems/day-of-the-year/solution/wei-rao-li-lun-mo-ni-ti-run-nian-pan-dua-8yzu/

                                        class Solution {
public:
    int dayOfYear(string date) {
        int days[12] = {31,28,31,30,31,30,31,31,30,31,30,31};
        
        string yy = date.substr(0, 4);
        string mm = date.substr(5, 2);
        string dd = date.substr(8, 2);

        int y = stoi(yy);
        int m = stoi(mm);
        int d = stoi(dd);

        int ans = 0;
        if ( ( (0==y%4) && (0!=y%100)) || ( 0==y%400) ) 
        {
            days[1]++;
        }

        for (int i = 0; i < m - 1; i++) {
            ans += days[i];
        }
        ans += d;

        return ans;
    }
};

                                        # 6.747. 至少是其他数字两倍的最大数 (opens new window)-

                                        • //学习这种，寻找最大值和次大值的遍历方法！
                                        class Solution {
public:
    int dominantIndex(vector<int>& nums) {
        int m1 = -1, m2 = -1;
        int index = -1;

        //学习这种，寻找最大值和次大值的遍历方法！
        for (int i = 0; i < nums.size(); i++) {
            if (nums[i] > m1) {
                m2 = m1;
                m1 = nums[i];
                index = i;
            } else if (nums[i] > m2) {
                m2 = nums[i];
            }
        }
        return m1 >= m2 * 2 ? index : -1;
    }
};

                                        # 7.阅读理解-1656. 设计有序流 (opens new window)

                                        看不懂题但是按照题目的文字写

                                        • 换句话说是插入位置永远固定，按坐标

                                        • 但是返回值要按照当时的ptr来判断，如果插入位置刚好在ptr，才移动ptr。

                                        class OrderedStream {
private:
    vector<string> inner_ds;
    int ptr;
public:
    OrderedStream(int n) {
        inner_ds.resize( n+1 );
        ptr=1;
    }
    
    vector<string> insert(int idKey, string value) {
        inner_ds[idKey]=value;

        vector<string> res;
        while( inner_ds.size()>ptr && !inner_ds[ptr].empty() )
        {
            res.push_back( inner_ds[ptr] );
            ++ptr;
        }
        return res;
    }
};

/**

 * OrderedStream* obj = new OrderedStream(n);
 * vector<string> param_1 = obj->insert(idKey,value);
 */

                                        # 9.数学技巧-斜率变化-1232. 缀点成线 (opens new window)

                                        class Solution {
public:
    bool checkStraightLine(vector<vector<int>>& coordinates) {

        //为了避免，垂直于x和y轴的，并且防止/取整，我们将斜率变化
        for(int index=2; index<coordinates.size(); ++index)
        {
            /*
            coordinates[index][1]-coordinates[index-1][1] 
            / coordinates[index][0]-coordinates[index-1][0]

            coordinates[1][1]-coordinates[0][1]
            /coordinates[1][0]-coordinates[0][0];
            */
            if( 
        ( coordinates[index][1]-coordinates[index-1][1] ) * ( coordinates[1][0]-coordinates[0][0]  )
        !=
        ( coordinates[index][0]-coordinates[index-1][0] ) * ( coordinates[1][1]-coordinates[0][1] )
            )
            {
                return false;
            }

        }

        return true;
    }
};

//判断区间[start,end]是否满足递归
bool solve( vector<int> & postorder, int start, int end)
{
    //递归边界，表示完全符合
    if( start>=end )
    {
        return true; 
    }

    //最后end位置的显然是“根”
    int cur_index = start;

    //寻找，整个“左子树”
    while( postorder[cur_index] < postorder[end] )
    {
        ++cur_index;
    }

    int mid_index=cur_index;
    //寻找，整个“右子树”
    while( postorder[cur_index] > postorder[end] )
    {
        ++cur_index;
    }

    //判断cur_index == end写在前面，其实是1个减枝！
    return 
    cur_index == end 
    && solve( postorder, start, mid_index-1 ) 
    && solve( postorder, mid_index, end-1);
}
class Solution {
public:
    bool verifyPostorder(vector<int>& postorder) {

        return solve( postorder, 0, postorder.size()-1 );
    }
};

                                        class Solution {
public:
    bool exist(vector<vector<char>>& board, string word) {
        rows = board.size();
        cols = board[0].size();
        for(int i = 0; i < rows; i++) {
            for(int j = 0; j < cols; j++) {
                if(dfs(board, word, i, j, 0)) 
                {
                    return true;
                }
            }
        }
        return false;
    }
private:
    int rows, cols;
    bool dfs(vector<vector<char>>& board, string word, int i, int j, int k) {
        if(i >= rows || i < 0 || j >= cols || j < 0 || board[i][j] != word[k]) 
        {
            return false;
        }
        if(k == word.size() - 1) 
        {
            return true;
        }

        board[i][j] = '\0'; //标记访问过了⭐️⭐️⭐️
        bool res = dfs(board, word, i + 1, j, k + 1) || dfs(board, word, i - 1, j, k + 1) || 
                      dfs(board, word, i, j + 1, k + 1) || dfs(board, word, i , j - 1, k + 1);
        board[i][j] = word[k]; //重新放回来，准备回溯⭐️⭐️⭐️

        return res;
    }
};

class Solution {
    Node * pre_node_ptr, *head;
    void DFS(Node * root)
    {
        if( nullptr==root ) return ;

        DFS( root->left );//左边
        //根
        if( nullptr!=pre_node_ptr )
        {
            pre_node_ptr->right=root;
        }
        else
        {
            head=root;
        }

        root->left=pre_node_ptr;
        pre_node_ptr=root;

        DFS( root->right );//右边
    }
public:
    Node* treeToDoublyList(Node* root) {
        if( nullptr==root )
        {
            return root;
        }
        DFS( root );
        //构造循环
        head->left=pre_node_ptr;
        pre_node_ptr->right=head;
        return head; //循环，双向链表
    }
};

 *     int val;

 *     TreeNode *right;
 *     TreeNode() : val(0), left(nullptr), right(nullptr) {}
 *     TreeNode(int x) : val(x), left(nullptr), right(nullptr) {}
 *     TreeNode(int x, TreeNode *left, TreeNode *right) : val(x), left(left), right(right) {}

class Solution {
public:
    TreeNode* insertIntoMaxTree(TreeNode* root, int val) {
        if( nullptr==root )
        {
            TreeNode * res=new TreeNode(val);

        }

        if( root->val < val )

            TreeNode * res=new TreeNode(val);

        return root;
    }

                                        • 参考资料：https://leetcode.cn/problems/nge-tou-zi-de-dian-shu-lcof/solution/jian-zhi-offer-60-n-ge-tou-zi-de-dian-sh-z36d/
                                          • 使用的正向的动态规划，使用test的方式，解决了越界！
                                          • 此外，第1次使用memset给double赋值为0
                                        double dp[12][100];
double test(int i,int j)
{
    if( i<0 || j<0 )
    {
        return 0.0;
    }

    return dp[i][j];
}

void init()
{
    for(int index=1; index<=6; ++index)
    {
        dp[1][index]=1.0/6.0;
    }
}
class Solution {
public:
    vector<double> dicesProbability(int n) {
        memset(dp, 0, sizeof(dp) );
        //f(n,k)表示n个筛子弄出k点的概率为
        init();

        //开始动态规划
        for(int loop=2; loop<=n; ++loop)
        {
            for(int index=loop; index<=6*loop; ++index)
            {
                for(int i=1; i<=6; ++i)
                {
                    dp[loop][index]+=( 1.0/6.0 * test(loop-1, index-i) );
                }    
            }
        }

        vector<double> res;
        for(int index=n; index<=6*n; ++index)
        {
            res.push_back( dp[n][index] );
        }
        return res;
    }
};
