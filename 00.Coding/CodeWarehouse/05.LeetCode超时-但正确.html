    # 05.LeetCode超时-但正确的题目

    # 1.745. 前缀和后缀搜索 (opens new window)

    • 超时14/17
    class WordFilter {
    vector<string> inner_strs;
public:
    WordFilter(vector<string>& words) {
        inner_strs=words;
    }
    
    int f(string pref, string suff) {
        int res=inner_strs.size()-1;
        for(; res>=0; --res)
        {
            if( inner_strs[res].size() < pref.size() || inner_strs[res].size() < suff.size() )
            {
                continue;
            }
            
            if( 0==inner_strs[res].find(pref) 
            && inner_strs[res].size()-suff.size() == inner_strs[res].rfind(suff) )
            {
                return res;
            }

        }
        return res;
    }
};

/**
 * Your WordFilter object will be instantiated and called as such:
 * WordFilter* obj = new WordFilter(words);
 * int param_1 = obj->f(pref,suff);
 */
