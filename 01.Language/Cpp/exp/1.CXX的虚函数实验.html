          # 2.基类pointer指向派生类

          #include<iostream>
using namespace std;

class first
{
	public:
		void demo()
		{
			cout<<"demo"<<endl;
			test();
		}
		virtual void test()
		{
			cout<<"first virtual"<<endl;
		}
};

class second:public first
{
	public:
		virtual void test()
		{
			cout<<"second virtual"<<endl;
		}
};

int main()
{
	first * fp=nullptr;
	second a;
  fp=&a; 				//fp = static_cast<first *>(&a);
	fp->test();   //fp->vptr->test()   fp指针的步长和first是一样的，但是由于vptr都在class的第1个位置，所以在往指向的second的object进行偏移的时候，走的还是second的虚函数
	
	return 0;
}

          ​ 原因，class A中非成员函数A::mytest()在class外边，而class B中首先继承了对A::mytest()的继承权，但是显然也不在class B里面，然后弄了个虚函数，那么class B的最开始内存布局上就多了一个vptr，指向一个虚函数表（里面放的才是虚函数）

          ​ 基类指针指向派生类，然后调用mytest，但是由于基类指针寻找他原先class A中mytest的方式没有vptr可以偏移到，而是简单的去找A::mytest()在哪，因而，现在这样调用，也还是调用的A::mytest()。

          # 不存在虚成员变量

          #include<iostream>
using namespace std ;

class A
{
	public:
		A() { m_c = 'A' ; }
		char m_c ;
} ;

class B : public A
{
	public:
		B() { m_c = 'B' ; }
		char m_c ;
} ;

int main()
{
	A a ;
	B b ;

	A * p = &b ;
	cout << p->m_c << endl ; //p是A类型，步长是A，偏移到B中的继承下来的A的m——c的内存布局上
	cout << ((B*)p)->m_c << endl ; //修改p为B类型，步长是B，偏移到B中的新的B的m——c的内存布局上
	return 0 ;
}
